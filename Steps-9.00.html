Create New Measures 9

 Calculated Measures:

To create the measures  follow the instructions

1. Right-click on the hotel booking Table and select New measure

2. Type the functions below and press enter   

     1. DailyRate= Average(hotel_bookings[Daily Rate)  - This is the Average Daily Rate

     2. DailyRate YoY% =

      IF(

            ISFILTERED( DateTable[Date]),

                 ERROR("Time intelligence quick measures can only be grouped or filtered by the Power BI-provided date hierarchy primary  date column."),

                     VAR __PREV_YEAR =

                              CALCULATE(

                                      [.DailyRate],

                                                     DATEADD('DateTable'[Date], -1, YEAR)

        )

                                RETURN

                                              DIVIDE([.DailyRate] - __PREV_YEAR, __PREV_YEAR)+0

 

3. Repeated Guest = SUM(hotel_bookings[is_repeated_guest])

4. AvgRepeated-guest = AVERAGE(hotel_bookings[is_repeated_guest])

5. Repeated YoY% =

      IF(

           ISFILTERED(DateTable[Date]),

             ERROR("Time intelligence quick measures can only be grouped or filtered by the Power BI-provided date hierarchy or primary date  column."),

                     VAR __PREV_YEAR =

                                   CALCULATE(

                                          [AvgRepeated-guest],

                                                    DATEADD('DateTable'[Date], -1, YEAR)

        )

                                      RETURN

                                             DIVIDE([AvgRepeated-guest] - __PREV_YEAR, __PREV_YEAR)+0

 

6 . ResortHotels = COUNTROWS(FILTER('hotel_bookings','hotel bookings'[hotel] = "Resort Hotel"))

7.  CityHotel = COUNTROWS(FILTER('hotel_bookings','hotel bookings'[hotel] = "City Hotel"))

8.  TotalBooking = COUNT(hotel bookings[hotel])

9.  1KPIDailyRate _Color = SWITCH(TRUE(),hotel bookings[.Daily Rate YoY%]>=3.0,"#ff0049",[.Daily Rate YoY%]<3.0,"#418b41")

10.  2ArrColor-Daily = SWITCH (TRUE (), [.Daily Rate YoY%] >= 0.0000, UNICHAR (9650)[.Daily Rate YoY%] < 0.0000,

          UNICHAR (9660))

11. 3COLOR-DAILYRATE-YOY = SWITCH (TRUE (), [.Daily Rate YoY%] >= 0.0000,"#418b41",

                                 [.Daily Rate YoY%] < 0.0000, "#ff0049")

 

12.  4KPIRepted _Color = SWITCH(TRUE(),hotel bookings[.RepeatedYoY%]>=3.0,"#ff0049", [.Repeated YoY%]<3.0,"#418b41")

13.  5KPIREPEATED_YOY_Color = SWITCH (TRUE (), [.Repeated YoY%] >= 0.0000,"#418b41",

          [.Repeated YoY%] < 0.0000, "#ff0049")

14. 6ArrColorRepted = SWITCH (TRUE (), [.Repeated YoY%] >= 0.0000, UNICHAR (9650),
